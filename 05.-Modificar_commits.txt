
# EJEMPLO: SI TIENES 5 COMMITS PERO OLVIDASTE AGREGAR ALGO AL ULTIMO QUE HICISTE Y NO QUIERES AGREGAR OTRO OTRO. ANTES TIENES QUE AGREGAR LOS ARCHIVOS FALTANTES AL STAGE AREA 
git commit --amend


# SOLO QUIERES MODIFICAR EL MENSAJE PUEDES USAR ESTE COMANDO 
git commit --amend


# SI EN EL MISMO ESCENARIO QUIERO MODIFICAR EL MENSAJE DEL COMMIT 3 ESTANDO EN EL 5 (NO RECOMENDADO)
git rebase -i head~3
git commit --amend
git rebase --continue

# DESHACER UN COMMIT (LLEVAR EL HEAD AL COMMIT DESEADO) DE ESTA MANERA TODO LO QUE HABIA EN EL O LOS COMMITS ANTERIORES LO DEJA EN EL STAGE AREA
# ESTO PERMITE QUE SI SE OLVIDO AGREGAR ALGO PUEDAS AGREGAR LO QUE FALTO Y VOLVER HACER EL COMMIT
# CUANDO USAS SOFT LOS ARCHIVOS QUE TIENES EN STAGEA AREA SE MANTIENEN Y AGREGA LOS DEMAS
git reset --soft ID
git reset --soft head~1 (ESTO LO QUE HACE ES POSICIONARTE UNO ATRAS DEL HEAD OSEA EL PUNTERO ACTUAL SE ELIMINARIA SI ES EL NUMERO 1)


# CON ESTE COMANDO SI LIMPIA EL STAGE AREA Y NO MODIFICA TU AREA DE TRABAJO 
git --reset --mixed ID


# ESTE COMANDO BORRA EL ULTIMO COMMIT Y DESCARTA TODO LO QUE HAY EN STAGE AREA, PONE LOS ARCHIVOS DEL COMMIT BORRADO EN EL AREA DE TRABAJO 
git --reset --hard
